  <script>
    const API_BASE = 'http://localhost:3000/api';

    const App = {
      currentUser: null,
      currentProduct: null,

      init() {
        this.loadUser();
        this.updateNav();
        this.navigate('home');
        this.loadProducts();
      },

      loadUser() {
        const userStr = localStorage.getItem('current_user');
        if (userStr) {
          this.currentUser = JSON.parse(userStr);
        }
      },

      updateNav() {
        const nav = document.getElementById('auth-nav');
        if (this.currentUser) {
          nav.innerHTML = `
            <button onclick="App.navigate('publish')">发布商品</button>
            <button onclick="App.navigate('user'); App.showUserTab('orders')">我的订单</button>
            <button onclick="App.navigate('user')">${this.currentUser.nickname}</button>
            <button onclick="App.logout()">退出</button>
          `;
        } else {
          nav.innerHTML = `
            <button onclick="App.showModal('login-modal')">登录</button>
            <button onclick="App.showModal('register-modal')">注册</button>
          `;
        }
      },

      navigate(page) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
        document.getElementById('page-' + page).classList.add('active');
      },

      showModal(id) {
        document.getElementById(id).classList.add('active');
      },

      async loadProducts() {
        try {
          const keyword = document.querySelector('.search-input').value.toLowerCase();
          const category = document.querySelector('.category-select').value;

          const res = await fetch(`${API_BASE}/products`);
          let products = await res.json();

          if (keyword) {
            products = products.filter(p =>
              p.title.toLowerCase().includes(keyword) ||
              (p.description && p.description.toLowerCase().includes(keyword))
            );
          }
          if (category) {
            products = products.filter(p => p.category === category);
          }

          products.sort((a, b) => b.id - a.id);

          const userRes = await fetch(`${API_BASE}/users`);
          const users = await userRes.json();
          const userMap = {};
          users.forEach(u => userMap[u.id] = u);

          const icons = { '手机数码': '📱', '图书教材': '📚', '交通工具': '🚲', '生活用品': '💡', '运动器材': '🏀', '其他': '📦' };

          const container = document.querySelector('.products-grid');
          if (products.length === 0) {
            container.innerHTML = `
              <div class="empty-state" style="grid-column: 1/-1;">
                <div class="icon">📦</div>
                <p>暂无商品</p>
              </div>
            `;
            return;
          }

          container.innerHTML = products.map(p => {
            const seller = userMap[p.seller_id] || {};
            return `
              <div class="product-card" onclick="App.showDetail(${p.id})">
                <div class="image">${icons[p.category] || '📦'}</div>
                <div class="info">
                  <div class="title">${p.title}</div>
                  <div>
                    <span class="price">¥${p.price}</span>
                    ${p.original_price ? `<span class="original-price">¥${p.original_price}</span>` : ''}
                  </div>
                  <div class="meta">
                    <span>👤 ${seller.nickname || '未知'}</span>
                    <span class="campus">🏫 ${seller.campus || ''}</span>
                  </div>
                  <div style="margin-top:8px;">
                    <span class="status-badge ${p.status === 'available' ? 'status-available' : 'status-sold'}">
                      ${p.status === 'available' ? '在售' : '已售'}
                    </span>
                  </div>
                </div>
              </div>
            `;
          }).join('');
        } catch (err) {
          console.error('加载商品失败:', err);
          this.showToast('加载商品失败', 'error');
        }
      },

      async showDetail(productId) {
        try {
          const res = await fetch(`${API_BASE}/products/${productId}`);
          this.currentProduct = await res.json();
          if (!this.currentProduct) return;

          const userRes = await fetch(`${API_BASE}/users`);
          const users = await userRes.json();
          const seller = users.find(u => u.id === this.currentProduct.seller_id) || {};

          document.getElementById('detail-title').textContent = this.currentProduct.title;
          document.getElementById('detail-price').innerHTML = `¥${this.currentProduct.price}${this.currentProduct.original_price ? `<span class="original-price">¥${this.currentProduct.original_price}</span>` : ''}`;
          document.getElementById('detail-desc').textContent = this.currentProduct.description || '暂无描述';
          document.getElementById('detail-category').textContent = this.currentProduct.category || '未分类';
          document.getElementById('detail-seller').textContent = seller.nickname || '未知';
          document.getElementById('detail-phone').textContent = seller.phone || '未填写';
          document.getElementById('detail-campus').textContent = seller.campus || '未填写';

          const icons = { '手机数码': '📱', '图书教材': '📚', '交通工具': '🚲', '生活用品': '💡', '运动器材': '🏀', '其他': '📦' };
          document.getElementById('detail-image').textContent = icons[this.currentProduct.category] || '📦';

          const buyBtn = document.getElementById('buy-btn');
          if (this.currentProduct.status !== 'available' || !this.currentUser || this.currentProduct.seller_id === this.currentUser.id) {
            buyBtn.style.display = 'none';
          } else {
            buyBtn.style.display = 'block';
          }

          this.navigate('detail');
        } catch (err) {
          console.error('加载商品详情失败:', err);
          this.showToast('加载商品详情失败', 'error');
        }
      },

      async handleLogin(e) {
        e.preventDefault();
        const username = document.getElementById('login-username').value;
        const password = document.getElementById('login-password').value;

        try {
          const res = await fetch(`${API_BASE}/login`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ username, password })
          });
          const data = await res.json();

          if (!data.success) {
            this.showToast(data.message || '登录失败', 'error');
            return false;
          }

          this.currentUser = data.user;
          localStorage.setItem('current_user', JSON.stringify(this.currentUser));
          this.updateNav();
          document.getElementById('login-modal').classList.remove('active');
          this.showToast('登录成功', 'success');
        } catch (err) {
          console.error('登录失败:', err);
          this.showToast('网络错误，登录失败', 'error');
        }
        return false;
      },

      async handleRegister(e) {
        e.preventDefault();
        const username = document.getElementById('reg-username').value;
        const password = document.getElementById('reg-password').value;
        const nickname = document.getElementById('reg-nickname').value || username;
        const phone = document.getElementById('reg-phone').value;
        const campus = document.getElementById('reg-campus').value;

        try {
          const res = await fetch(`${API_BASE}/register`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ username, password, nickname, phone, campus })
          });
          const data = await res.json();

          if (!data.success) {
            this.showToast(data.message || '注册失败', 'error');
            return false;
          }

          this.currentUser = data.user;
          localStorage.setItem('current_user', JSON.stringify(this.currentUser));
          this.updateNav();
          document.getElementById('register-modal').classList.remove('active');
          this.showToast('注册成功', 'success');
        } catch (err) {
          console.error('注册失败:', err);
          this.showToast('网络错误，注册失败', 'error');
        }
        return false;
      },

      logout() {
        this.currentUser = null;
        localStorage.removeItem('current_user');
        this.updateNav();
        this.navigate('home');
        this.showToast('已退出登录', 'success');
      },

      async handlePublish(e) {
        e.preventDefault();
        if (!this.currentUser) {
          this.showToast('请先登录', 'error');
          return false;
        }

        const title = document.getElementById('pub-title').value;
        const description = document.getElementById('pub-desc').value;
        const price = parseFloat(document.getElementById('pub-price').value);
        const original_price = document.getElementById('pub-original-price').value ? parseFloat(document.getElementById('pub-original-price').value) : null;
        const category = document.getElementById('pub-category').value;

        try {
          const res = await fetch(`${API_BASE}/publish`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              seller_id: this.currentUser.id,
              title,
              description,
              price,
              original_price,
              category
            })
          });
          const data = await res.json();

          if (data.success) {
            document.getElementById('publish-form').reset();
            this.showToast('发布成功', 'success');
            this.navigate('home');
            this.loadProducts();
          } else {
            this.showToast('发布失败', 'error');
          }
        } catch (err) {
          console.error('发布商品失败:', err);
          this.showToast('网络错误，发布失败', 'error');
        }
        return false;
      },

      async handleBuy() {
        if (!this.currentUser) {
          this.showToast('请先登录', 'error');
          return;
        }

        if (!this.currentProduct || this.currentProduct.seller_id === this.currentUser.id) {
          this.showToast('不能购买自己的商品', 'error');
          return;
        }

        if (this.currentProduct.status !== 'available') {
          this.showToast('商品已售出', 'error');
          return;
        }

        if (!confirm('确认购买此商品？')) return;

        try {
          const res = await fetch(`${API_BASE}/buy`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              product_id: this.currentProduct.id,
              buyer_id: this.currentUser.id
            })
          });
          const data = await res.json();

          if (data.success) {
            this.showToast('购买成功，请联系卖家', 'success');
            this.showDetail(this.currentProduct.id);
          } else {
            this.showToast(data.message || '购买失败', 'error');
          }
        } catch (err) {
          console.error('购买商品失败:', err);
          this.showToast('网络错误，购买失败', 'error');
        }
      },

      async showUserTab(tab) {
        const content = document.getElementById('user-content');

        document.querySelectorAll('.sidebar .menu-item').forEach((item, index) => {
          item.classList.remove('active');
          if ((tab === 'info' && index === 0) || (tab === 'my-products' && index === 1) || (tab === 'orders' && index === 2)) {
            item.classList.add('active');
          }
        });

        if (!this.currentUser) return;

        try {
          if (tab === 'info') {
            const productsRes = await fetch(`${API_BASE}/products`);
            const products = await productsRes.json();
            const myProducts = products.filter(p => p.seller_id === this.currentUser.id);
            const available = myProducts.filter(p => p.status === 'available').length;
            const sold = myProducts.filter(p => p.status === 'sold').length;

            content.innerHTML = `
              <h3>个人信息</h3>
              <p><strong>用户名：</strong>${this.currentUser.username}</p>
              <p><strong>昵称：</strong>${this.currentUser.nickname}</p>
              <p><strong>手机：</strong>${this.currentUser.phone || '未设置'}</p>
              <p><strong>学校：</strong>${this.currentUser.campus || '未设置'}</p>
              <hr style="margin: 20px 0; border: none; border-top: 1px solid var(--border)">
              <h4>商品统计</h4>
              <p>总发布：<strong>${myProducts.length}</strong> 件</p>
              <p>在售：<strong style="color: var(--success)">${available}</strong> 件</p>
              <p>已售：<strong style="color: var(--danger)">${sold}</strong> 件</p>
            `;
          } else if (tab === 'my-products') {
            const productsRes = await fetch(`${API_BASE}/products`);
            const products = await productsRes.json();
            const myProducts = products.filter(p => p.seller_id === this.currentUser.id).sort((a, b) => b.id - a.id);

            if (myProducts.length === 0) {
              content.innerHTML = '<h3>我的商品</h3><div class="empty-state"><p>暂无商品</p></div>';
              return;
            }

            content.innerHTML = `
              <h3>我的商品</h3>
              <div class="orders-list">
                ${myProducts.map(p => `
                  <div class="order-item">
                    <div class="order-info">
                      <h4>${p.title}</h4>
                      <p>¥${p.price} - ${p.category}</p>
                    </div>
                    <span class="status-badge ${p.status === 'available' ? 'status-available' : 'status-sold'}">
                      ${p.status === 'available' ? '在售' : '已售'}
                    </span>
                  </div>
                `).join('')}
              </div>
            `;
          } else if (tab === 'orders') {
            const ordersRes = await fetch(`${API_BASE}/orders`);
            const orders = await ordersRes.json();
            const myOrders = orders.filter(o => o.buyer_id === this.currentUser.id || o.seller_id === this.currentUser.id).sort((a, b) => b.id - a.id);

            const productsRes = await fetch(`${API_BASE}/products`);
            const products = await productsRes.json();
            const productMap = {};
            products.forEach(p => productMap[p.id] = p);

            if (myOrders.length === 0) {
              content.innerHTML = '<h3>我的订单</h3><div class="empty-state"><p>暂无订单</p></div>';
              return;
            }

            content.innerHTML = `
              <h3>我的订单</h3>
              <div class="orders-list">
                ${myOrders.map(o => {
                  const product = productMap[o.product_id] || {};
                  const isBuyer = o.buyer_id === this.currentUser.id;
                  return `
                    <div class="order-item">
                      <div class="order-info">
                        <h4>${product.title || '未知商品'}</h4>
                        <p>¥${product.price || 0} - ${o.created_at} - ${isBuyer ? '我购买的' : '我出售的'}</p>
                      </div>
                      <span class="status-badge ${o.status === 'pending' ? 'status-available' : 'status-sold'}">
                        ${o.status === 'pending' ? '待完成' : '已完成'}
                      </span>
                    </div>
                  `;
                }).join('')}
              </div>
            `;
          }

          document.getElementById('user-nickname').textContent = this.currentUser.nickname;
          document.getElementById('user-campus').textContent = '🏫 ' + (this.currentUser.campus || '未设置');
          document.getElementById('user-avatar').textContent = this.currentUser.nickname.charAt(0);
        } catch (err) {
          console.error('加载用户数据失败:', err);
          this.showToast('加载数据失败', 'error');
        }
      },

      showToast(message, type = 'success') {
        const toast = document.createElement('div');
        toast.className = `toast active ${type}`;
        toast.textContent = message;
        document.body.appendChild(toast);
        setTimeout(() => {
          toast.classList.remove('active');
          setTimeout(() => toast.remove(), 300);
        }, 2000);
      }
    };

    document.addEventListener('DOMContentLoaded', () => App.init());

    document.querySelectorAll('.modal').forEach(modal => {
      modal.addEventListener('click', (e) => {
        if (e.target.classList.contains('modal')) {
          modal.classList.remove('active');
        }
      });
    });
